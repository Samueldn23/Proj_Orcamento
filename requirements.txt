# Web Framework
flet                    # Framework Flutter/Python para criar interfaces gráficas multiplataforma
flet-core               # Componentes principais do Flet
flet-desktop            # Suporte específico para aplicações desktop com Flet

# Banco de Dados
SQLAlchemy              # ORM (Object Relational Mapper) para trabalhar com bancos de dados
psycopg2                # Driver PostgreSQL para Python
supabase                # Cliente Python para Supabase (Backend as a Service)
postgrest               # Cliente Python para PostgreSQL RESTful API

# Configuração e Cache
python-dotenv           # Carrega variáveis de ambiente de arquivos .env
cachetools              # Implementações de cache em memória (usado para otimizar consultas)

# Utilitários
pydantic                # Validação de dados e gerenciamento de configurações usando type hints
python-dateutil         # Extensões úteis para manipulação de datas e horários
typing-extensions       # Recursos adicionais para type hints
watchdog                # Monitora mudanças em arquivos do sistema (útil para desenvolvimento)

# HTTP e Comunicação
httpx                   # Cliente HTTP moderno e assíncrono
websockets              # Implementação do protocolo WebSocket
requests                # Biblioteca HTTP para fazer requisições web

# Formatação e Tipos
types-python-dateutil   # Type hints para python-dateutil (auxilia no desenvolvimento)

# Opcional - para desenvolvimento
ipython                 # Shell interativo melhorado para desenvolvimento
